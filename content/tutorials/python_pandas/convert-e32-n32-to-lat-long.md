```python
df = df.dropna()

def getUTMs(row):
    tup = utm.to_latlon(row.iloc[0],row.iloc[1], 32, 'U')
    return pd.Series(tup[:2])

df[['lat','long']] = df[['e32','n32']].apply(getUTMs , axis=1)
```
